       19. HOVER EFFECTS

    /* Standard hover lift effect for interactive cards */
    .hover-lift {
      @apply transition-all duration-200 ease-out;
    }

    .hover-lift:hover {
      @apply -translate-y-1 shadow-lg;
    }

    /* Subtle hover for buttons/links */
    .hover-subtle {
      @apply transition-all duration-150 ease-out;
    }

    .hover-subtle:hover {
      @apply opacity-90;
    }

    /* Hover with background change */
    .hover-bg {
      @apply transition-colors duration-150 ease-out;
    }

    .hover-bg:hover {
      @apply bg-gray-50;
    }

    /* Hover with border highlight */
    .hover-border {
      @apply transition-all duration-150 ease-out border border-transparent;
    }

    .hover-border:hover {
      @apply border-primary-300;
    }

    /* Scale on hover (for icons/avatars) */
    .hover-scale {
      @apply transition-transform duration-150 ease-out;
    }

    .hover-scale:hover {

    }
